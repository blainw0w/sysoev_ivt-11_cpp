struct book
{
	char* namebook = NULL;

};

void sortalf(char** data, int lenD)
{
	int j = 0;
	char* tmp = 0;
	for (int i = 0; i < lenD; i++)
	{
		j = i;
		for (int k = i; k < lenD; k++) {
			if (strcmp(data[j], data[k]) > 0) {
				j = k;
			}
		}
		tmp = data[i];
		data[i] = data[j];
		data[j] = tmp;
	}
}

void authorsbooks(char* q, char* w, book* book, int N)
{
	int u = 0;
	cout << endl << "Список книг автора: " << endl;
	for (int i = 0; i < N; i++)
	{
		if (strcmp(q, book[i].surname) == 0 && strcmp(w, book[i].name) == 0) { u++; cout << u << ") " << book[i].namebook << endl; }
	}
	if (u == 0) cout << "Книг интересующего вас автора в библиотеке нет" << endl;
}

void pubhousewriters(char* m, book* book, int N)
{

	for (int i = 0; i < N; i++)
	{
		if (strcmp(m, book[i].pubhouse) == 0) { surnames1[i] = book[i].surname; u++; }
		else surnames1[i] = NULL;
	}
	char** surnames2 = new char* [u]; int j = 0;
	for (int i = 0; i < N; i++)
	{
		if (surnames1[i] == NULL) { j = j; }
		else { surnames2[j] = surnames1[i]; j++; }
	}
	sortalf(surnames2, u);
	if (u == 0) cout << "Писателей интересующего вас издательства в библиотеке нет" << endl;
	else cout << endl << "Список фамилий писателей издательства: " << endl;
	for (int i = 0; i < u; i++)
	{
		cout << i + 1 << ") " << surnames2[i] << endl;
	}
}
